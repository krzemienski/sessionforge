```env
# =============================================================================
# SessionForge Environment Variables
# =============================================================================
# Copy this file to .env and fill in your values.
#
# Deployment modes:
#   Docker dev:    docker compose up (uses bundled Postgres, no Redis)
#   Self-hosted:   docker compose -f docker-compose.self-hosted.yml up (Postgres + Redis)
#   Cloud/Vercel:  Vercel dashboard env vars (Neon Postgres, Upstash Redis)
#
# Generate secrets with: openssl rand -base64 32
# =============================================================================

# =============================================================================
# REQUIRED -- The app will not start without these
# =============================================================================

# PostgreSQL connection string.
# Docker dev:    postgresql://sessionforge:sessionforge@localhost:5432/sessionforge
#                (already set in docker-compose.yml -- you do NOT need this in .env for Docker)
# Self-hosted:   postgresql://user:password@your-postgres-host:5432/sessionforge
# Vercel/Neon:   postgresql://neondb_owner:PASSWORD@ep-xxx-pooler.region.aws.neon.tech/neondb?sslmode=require
DATABASE_URL=postgresql://sessionforge:sessionforge@localhost:5432/sessionforge

# Random secret used to sign auth tokens and sessions.
# Generate with: openssl rand -base64 32
BETTER_AUTH_SECRET=change-me-generate-with-openssl-rand-base64-32

# The public-facing URL of your deployment (no trailing slash).
# Used for OAuth redirects, email links, and CORS.
# Docker dev / local: http://localhost:3000
# Production: https://your-domain.com
NEXT_PUBLIC_APP_URL=http://localhost:3000

# Same as NEXT_PUBLIC_APP_URL -- used internally by better-auth for URL validation.
BETTER_AUTH_URL=http://localhost:3000

# =============================================================================
# DATABASE -- Driver selection
# =============================================================================

# Force the database driver instead of auto-detection from the DATABASE_URL.
# Options: "postgres" (standard postgres-js) | "neon" (Neon serverless HTTP)
# Docker/self-hosted: set to "postgres" (required for standard PostgreSQL)
# Vercel/Neon: omit or set to "neon" (auto-detected from neon.tech URL)
# Already set in docker-compose.yml -- you do NOT need this in .env for Docker.
DATABASE_DRIVER=postgres

# =============================================================================
# CACHE -- Redis (optional, choose one provider)
# =============================================================================
# Redis is optional. When not configured, the app runs without caching.
# The self-hosted compose file bundles a Redis container automatically.

# Standard Redis connection URL (ioredis-compatible).
# Use for self-hosted Redis (Docker Compose, bare metal, etc.)
# Example: redis://localhost:6379  or  redis://:password@redis:6379
# REDIS_URL=redis://localhost:6379

# --- OR Upstash Redis (managed serverless, required for Vercel/edge) ---
# Sign up at https://upstash.com and create a Redis database.
# UPSTASH_REDIS_URL=https://your-database.upstash.io
# UPSTASH_REDIS_TOKEN=your-upstash-redis-token

# =============================================================================
# QUEUE -- Background jobs and automation (optional)
# =============================================================================

# Secret for authenticating cron/webhook requests to /api/cron/automation.
# Generate with: openssl rand -base64 32
# CRON_SECRET=change-me-generate-with-openssl-rand-base64-32

# --- OR Upstash QStash (managed queue, required for Vercel/edge) ---
# Sign up at https://upstash.com and create a QStash instance.
# UPSTASH_QSTASH_TOKEN=your-qstash-token
# UPSTASH_QSTASH_CURRENT_SIGNING_KEY=sig_your-current-signing-key
# UPSTASH_QSTASH_NEXT_SIGNING_KEY=sig_your-next-signing-key

# =============================================================================
# AUTH -- OAuth providers (optional)
# =============================================================================

# GitHub OAuth App credentials for GitHub login.
# Create an OAuth App at: https://github.com/settings/developers
# Callback URL: http://localhost:3000/api/auth/callback/github (dev)
#               https://your-domain.com/api/auth/callback/github (prod)
# GITHUB_CLIENT_ID=your-github-client-id
# GITHUB_CLIENT_SECRET=your-github-client-secret

# =============================================================================
# BILLING -- Stripe (optional)
# =============================================================================

# Stripe secret key -- use sk_test_... for staging, sk_live_... for production.
# Find at: https://dashboard.stripe.com/apikeys
# STRIPE_SECRET_KEY=sk_test_your-stripe-secret-key

# Stripe webhook signing secret -- validates incoming webhook events.
# Create webhook at: https://dashboard.stripe.com/webhooks
# Endpoint: https://your-domain.com/api/webhooks/stripe
# STRIPE_WEBHOOK_SECRET=whsec_your-stripe-webhook-secret

# =============================================================================
# AI -- Claude Agent SDK (optional)
# =============================================================================

# Set to "true" to disable all AI agent features (session scanning, generation).
# When disabled, AI routes return graceful errors; editing/publishing still work.
# The Agent SDK uses CLI-inherited auth -- no API key needed.
# Default: false (AI enabled if Claude CLI is installed and authenticated).
# Docker dev: defaults to "true" in docker-compose.yml.
# DISABLE_AI_AGENTS=true

# =============================================================================
# OBSERVABILITY -- Deployment validation (optional)
# =============================================================================

# Bearer token for GET /api/deployment/validate.
# When set, unauthenticated requests return 401.
# Leave unset for open health/readiness checks.
# DEPLOYMENT_VALIDATE_TOKEN=change-me-generate-with-openssl-rand-base64-32
```
